<?php
session_start();

// Login fixo
$usuarioAdmin = "admin";
$senhaAdmin = "1234";

// Checa login
if(isset($_POST['login'])){
    if($_POST['usuario'] === $usuarioAdmin && $_POST['senha'] === $senhaAdmin){
        $_SESSION['logado'] = true;
    } else {
        $erro = "Usuário ou senha incorretos!";
    }
}

// Logout
if(isset($_GET['logout'])){
    session_destroy();
    header("Location: index.php");
    exit;
}

// Pastas de upload
$musicasDir = __DIR__ . '/musicas/';
$capasDir = __DIR__ . '/capas/';

// Cria pastas se não existirem
if(!file_exists($musicasDir)) mkdir($musicasDir, 0777, true);
if(!file_exists($capasDir)) mkdir($capasDir, 0777, true);

// Arquivo JSON
$dataFile = 'data.json';
if(!file_exists($dataFile)){
    file_put_contents($dataFile, json_encode([]));
}
$musicas = json_decode(file_get_contents($dataFile), true);

// Upload de música
if(isset($_POST['add_musica']) && isset($_SESSION['logado'])){
    $titulo = $_POST['titulo'];
    $descricao = $_POST['descricao'];
    $genero = $_POST['genero'];

    // Upload capa
    $capa = "";
    if(isset($_FILES['capa']) && $_FILES['capa']['error'] == 0){
        $ext = pathinfo($_FILES['capa']['name'], PATHINFO_EXTENSION);
        $capa = "capas/".time()."_capa.".$ext;
        move_uploaded_file($_FILES['capa']['tmp_name'], $capa);
    }

    // Upload MP3
    $mp3 = "";
    if(isset($_FILES['mp3']) && $_FILES['mp3']['error'] == 0){
        $ext = pathinfo($_FILES['mp3']['name'], PATHINFO_EXTENSION);
        $mp3 = "musicas/".time()."_audio.".$ext;
        move_uploaded_file($_FILES['mp3']['tmp_name'], $mp3);
    }

    if($capa && $mp3){ // só adiciona se ambos existirem
        $musicas[] = [
            "titulo"=>$titulo,
            "descricao"=>$descricao,
            "genero"=>$genero,
            "capa"=>$capa,
            "mp3"=>$mp3
        ];
        file_put_contents($dataFile, json_encode($musicas));
    }

    header("Location: index.php");
    exit;
}

<script>
const player = document.getElementById('player');
const filtro = document.getElementById('filtroGenero');
const cards = document.querySelectorAll('.card');

function tocar(caminho){
    player.src = caminho;
    player.play();
}

// Filtro por gênero
filtro.addEventListener('change', ()=>{
    const genero = filtro.value;
    cards.forEach(c=>{
        if(genero==='todos' || c.dataset.genero===genero){
            c.style.display='block';
        } else {
            c.style.display='none';
        }
    });
});
</script>
