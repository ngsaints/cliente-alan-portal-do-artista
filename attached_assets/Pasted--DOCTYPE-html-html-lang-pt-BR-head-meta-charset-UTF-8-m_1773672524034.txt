<script>

let songs=[

{title:"Minha Música 1",src:"music1.mp3"},
{title:"Minha Música 2",src:"music2.mp3"},
{title:"Minha Música 3",src:"music3.mp3"}

]

let currentIndex=0

const player=document.getElementById("player")

const songsList=document.getElementById("songsList")

function renderSongs(){

songsList.innerHTML=""

songs.forEach((s,i)=>{

let div=document.createElement("div")

div.className="song"

div.innerHTML=`

<strong>${s.title}</strong>

<br>

<button onclick="playSong(${i})">▶️ Ouvir</button>

<button onclick="openInterestModal('${s.title}')">💼 Interesse nesta música</button>

`

songsList.appendChild(div)

})

}

function playSong(i){

currentIndex=i

player.src=songs[i].src

player.play()

}

player.addEventListener("ended",function(){

currentIndex++

if(currentIndex>=songs.length){

currentIndex=0

}

player.src=songs[currentIndex].src

player.play()

})

renderSongs()

function registerArtist(){

let nome=prompt("Nome do artista")

if(nome){

document.getElementById("artistName").innerText=nome

localStorage.setItem("artistName",nome)

alert("Artista cadastrado!")

}

}

if(localStorage.getItem("artistName")){

document.getElementById("artistName").innerText=localStorage.getItem("artistName")

}

function choosePlan(name,max,price){

alert("Plano escolhido: "+name+"\nLimite de músicas: "+max+"\nPreço: R$"+price)

}

let vipPassword=localStorage.getItem("vipPassword")||"0000"

function showVip(){

let senha=prompt("Digite a senha VIP")

if(senha==vipPassword){

let nova=prompt("Deseja trocar senha? Digite nova ou cancelar")

if(nova){

vipPassword=nova

localStorage.setItem("vipPassword",nova)

alert("Senha atualizada")

}

alert("Acesso VIP liberado")

}

else{

alert("Senha incorreta")

}

}

function openInterestModal(song=""){

document.getElementById("interestModal").style.display="block"

document.getElementById("musicInterest").value=song

}

function closeInterest(){

document.getElementById("interestModal").style.display="none"

}

const whatsapp="5521999999999"

function sendInterest(){

let musica=document.getElementById("musicInterest").value

let tipo=document.getElementById("interestType").value

let msg=document.getElementById("interestMessage").value

let texto=`Olá tenho interesse.\n\nMusica: ${musica}\nTipo: ${tipo}\nMensagem: ${msg}`

let url=`https://wa.me/${whatsapp}?text=${encodeURIComponent(texto)}`

window.open(url)

}
