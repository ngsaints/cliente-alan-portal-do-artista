<script>
// ====== Simulação de banco usando localStorage ======
let artistas = JSON.parse(localStorage.getItem("artistas")) || [];
let personalizacao = JSON.parse(localStorage.getItem("personalizacao")) || {};
let vipPassword = localStorage.getItem("vipPassword") || "0000";
let currentIndex = -1;

// ====== Funções de Artistas e Planos ======
function saveArtistas() { localStorage.setItem("artistas", JSON.stringify(artistas)); }
function addArtista(name,email,plan="Grátis",maxSongs=2){ 
artistas.push({name,email,plan,maxSongs,songs:[],vipPassword:"0000"}); 
saveArtistas(); 
alert(`Cadastro de ${name} realizado! Usuário liberado automaticamente no plano ${plan}.`);
renderSongsTable();
}

// Botão cadastro
document.getElementById("registerBtn").addEventListener("click",()=>{
const name=prompt("Nome do artista:");
const email=prompt("E-mail do artista:");
if(name && email) addArtista(name,email);
});

// ====== Renderização da tabela de músicas ======
const songsTableBody=document.getElementById("songsTable").querySelector("tbody");
function renderSongsTable(){
songsTableBody.innerHTML="";
artistas.forEach((a,i)=>{
a.songs.forEach((s,j)=>{
const tr=document.createElement("tr");
tr.innerHTML=`<td>${s.title}</td><td>${s.genre}</td>
<td><button class="edit-btn" onclick="editSong(${i},${j})">Editar</button></td>`;
songsTableBody.appendChild(tr);
});
});
}
function editSong(artistaIndex,songIndex){
const song=artistas[artistaIndex].songs[songIndex];
const newTitle=prompt("Editar título da música:",song.title);
if(newTitle){ 
artistas[artistaIndex].songs[songIndex].title=newTitle; 
saveArtistas();
renderSongsTable(); 
}
}

// ====== Personalização ======
const bannerInput=document.getElementById("bannerInput");
const bannerPreview=document.getElementById("bannerPreview");
const artistNameInput=document.getElementById("artistNameInput");
const fontSelect=document.getElementById("fontSelect");
const colorInput=document.getElementById("colorInput");

// Carregar personalização
if(personalizacao.name) artistNameInput.value=personalizacao.name;
if(personalizacao.font) document.body.style.fontFamily=personalizacao.font;
if(personalizacao.bgColor) document.body.style.backgroundColor=personalizacao.bgColor;
if(personalizacao.banner) bannerPreview.src=personalizacao.banner;

bannerInput.addEventListener("change",(e)=>{
const file=e.target.files[0];
if(file){
const reader=new FileReader();
reader.onload=function(event){ bannerPreview.src=event.target.result; }
reader.readAsDataURL(file);
}
});
document.getElementById("saveCustomBtn").addEventListener("click",()=>{
personalizacao={};
personalizacao.name=artistNameInput.value;
personalizacao.font=fontSelect.value;
personalizacao.bgColor=colorInput.value;
if(bannerPreview.src) personalizacao.banner=bannerPreview.src;
localStorage.setItem("personalizacao",JSON.stringify(personalizacao));
document.body.style.fontFamily=personalizacao.font;
document.body.style.backgroundColor=personalizacao.bgColor;
alert("Personalização salva com sucesso!");
});

// ====== Área VIP ======
function checkVipPassword(){
const input=document.getElementById("vipPassword").value;
const vipMessage=document.getElementById("vipMessage");
if(input===vipPassword){
vipMessage.style.color="green";
vipMessage.textContent="Acesso permitido! Bem-vindo à Área VIP.";
}else{
vipMessage.style.color="red";
vipMessage.textContent="Senha incorreta. Tente novamente.";
}
}
function changeVipPassword(){
const newPass=document.getElementById("vipNewPassword").value;
if(newPass){
vipPassword=newPass;
localStorage.setItem("vipPassword",vipPassword);
alert("Senha VIP alterada com sucesso!");
document.getElementById("vipNewPassword").value="";
}else alert("Digite uma nova senha para atualizar.");
}

// ====== Player com autoplay ======
const audioPlayer=document.getElementById("audioPlayer");
const songsListElem=document.getElementById("songsList");
function renderSongsPlayer(){
songsListElem.innerHTML="";
artistas.forEach((a)=>{
a.songs.forEach((s)=>{
const li=document.createElement("li");
li.dataset.src=s.src||"musica1.mp3";
li.textContent=s.title;
li.addEventListener("click",()=>playSongByLi(li));
songsListElem.appendChild(li);
});
});
}
function playSongByLi(li){
currentIndex=Array.from(songsListElem.children).indexOf(li);
audioPlayer.src=li.dataset.src;
audioPlayer.play();
}
audioPlayer.addEventListener("ended",()=>{
currentIndex++;
if(currentIndex>=songsListElem.children.length) currentIndex=0;
audioPlayer.src=songsListElem.children[currentIndex].dataset.src;
audioPlayer.play();
});
renderSongsPlayer();

// ====== Escolher plano ======
function choosePlan(name,maxSongs,price){
alert(`Plano "${name}" escolhido. Limite de ${maxSongs} músicas. Preço: R$ ${price}.\nIntegração de pagamento futura.`);
}
